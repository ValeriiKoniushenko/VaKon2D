70 void update()
71 {
72 if (isKeyPressed())
73 {
74 if (lastState_ != State::Pressed || isRepeatable_)
75 {
76 if (std::chrono::duration_cast<TimeT>(std::chrono::system_clock::now() - lastUpdate_) >= frequency_)
77 {
78 onAction.trigger();
79 lastUpdate_ = std::chrono::system_clock::now();
80 lastState_ = State::Pressed;
81 }
82 }
83 }
84 else
85 {
86 lastState_ = State::None;
87 }
88 }

93 }

103 }

106 {
107 isRepeatable_ = isRepeatable;
108 }
109
110 LambdaMulticastDelegate<void()> onAction;
111
112protected:
113 _NODISCARD virtual bool isKeyPressed() const = 0;
114
115protected:
116 std::string name_;

118 TimeT frequency_ = TimeT(50);
